function (records, candidates, bsp, SP) 
{
    phenoDF <- framePhenoRec(records, bsp)
    if (!any(candidates %in% phenoDF$id)) {
        crit <- runif(length(candidates))
    }
    else {
        crit <- iidPhenoEval(phenoDF)
        crit <- crit[candidates]
    }
    names(crit) <- candidates
    return(crit)
